/* Should work well enough on modern CPUs for testing */
static inline int boot_cpu_has(int flag)
{
	u32 eax = (flag & 0x20) ? 0x80000001 : 1;
	u32 ecx, edx;

	asm volatile("cpuid"
		     : "+a" (eax), "=d" (edx), "=c" (ecx)
		     : : "ebx");

	return ((flag & 0x80 ? ecx : edx) >> (flag & 31)) & 1;
}
