#define MMP2_DS_SHIFT		11
#define MMP2_SLEEP_MASK		0x38

#define MFPR_MMP2(a, r, f0, f1, f2, f3, f4, f5, f6, f7)		\
	{							\
		.name = #a,					\
		.pin = a,					\
		.mfpr = r,					\
		.func = {					\
			MMP2_MUX_##f0,				\
			MMP2_MUX_##f1,				\
			MMP2_MUX_##f2,				\
			MMP2_MUX_##f3,				\
			MMP2_MUX_##f4,				\
			MMP2_MUX_##f5,				\
			MMP2_MUX_##f6,				\
			MMP2_MUX_##f7,				\
		},						\
	}

#define GRP_MMP2(a, m, p)		\
	{ .name = a, .mux = MMP2_MUX_##m, .pins = p, .npins = ARRAY_SIZE(p), }
