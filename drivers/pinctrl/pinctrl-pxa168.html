#define PXA168_DS_SHIFT		11
#define PXA168_SLEEP_MASK	0x38

#define MFPR_168(a, r, f0, f1, f2, f3, f4, f5, f6, f7)		\
	{							\
		.name = #a,					\
		.pin = a,					\
		.mfpr = r,					\
		.func = {					\
			PXA168_MUX_##f0,			\
			PXA168_MUX_##f1,			\
			PXA168_MUX_##f2,			\
			PXA168_MUX_##f3,			\
			PXA168_MUX_##f4,			\
			PXA168_MUX_##f5,			\
			PXA168_MUX_##f6,			\
			PXA168_MUX_##f7,			\
		},						\
	}

#define GRP_168(a, m, p)		\
	{ .name = a, .mux = PXA168_MUX_##m, .pins = p, .npins = ARRAY_SIZE(p), }
