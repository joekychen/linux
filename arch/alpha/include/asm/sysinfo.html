#define SSI_IEEE_STATE_AT_SIGNAL	15
#define SSI_IEEE_IGNORE_STATE_AT_SIGNAL	16

#define SSIN_UACPROC			6

#define UAC_BITMASK			7

#define UAC_SIGBUS			4

#endif /* __ASM_ALPHA_SYSINFO_H */
